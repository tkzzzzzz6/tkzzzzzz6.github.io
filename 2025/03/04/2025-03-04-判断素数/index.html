//                 еҲӨж–ӯзҙ ж•°зҡ„ж•°пҝҪ?   иҢғеӣҙдёҠйҷҗN
void Eratprime(int* isprime, int upper_board) {

    int i = 0;
    int j = 0;
    //еҲқе§ӢеҢ–isprime
    for (i = 2; i <= upper_board; i++)
        isprime[i] = 1;


    for (i = 2; i < (int)sqrt(upper_board); i++) {
        if (isprime[i]) {
            isprime[i] = 1;
        }
        for (j = 2; i * j <= upper_board; j++) {//зҙ ж•°зҡ„nеҖҚпјҲn >= 2пјүдёҚжҳҜзҙ пҝҪ?
            isprime[i * j] = 0;
        }
    }

}

            {
                isPrime[i * Prime[j]] = false;//жҜҸдёҖдёӘзҙ ж•°зҡ„ i еҖҚпјҲi >= 2пјүйғҪдёҚжҳҜзҙ ж•°пјҢзҪ®дёәfalse

                //иҝҷдёӘжҳҜж¬§жӢүзӯӣжі•зҡ„ж ёеҝғпјҢе®ғеҸҜд»ҘеҮҸе°‘йқһзҙ ж•°зҪ®falseзҡ„йҮҚеӨҚзҺҮ
                //ж„Ҹд№үжҳҜе°ҶжҜҸдёҖдёӘеҗҲж•°пјҲйқһзҙ ж•°пјүжӢҶжҲҗ 2пјҲжңҖе°Ҹеӣ ж•°пјүдёҺжңҖеӨ§еӣ пҝҪ?зҡ„д№ҳпҝҪ?
                if (i % Prime[j] == 0)
                    break;
            }
        }
    }
